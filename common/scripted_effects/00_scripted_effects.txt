recalc_current_economy_situation_icon_frame = {
	if = {
		limit = {
			check_variable = { current_economy_situation_icon_frame < 0.10 }
		}
		set_variable = { current_economy_situation_icon_frame_ui = 1 }
	}
	else_if = {
		limit = { 
			check_variable = { current_economy_situation_icon_frame < 0.30 }
		}
		set_variable = { current_economy_situation_icon_frame_ui = 2 }
	}
	else_if = {
		limit = { 
			check_variable = { current_economy_situation_icon_frame < 0.40 }
		}
		set_variable = { current_economy_situation_icon_frame_ui = 3 }
	}
	else_if = {
		limit = { 
			check_variable = { current_economy_situation_icon_frame < 0.50 }
		}
		set_variable = { current_economy_situation_icon_frame_ui = 4 }
	}
	else_if = {
		limit = { 
			check_variable = { current_economy_situation_icon_frame < 0.60 }
		}
		set_variable = { current_economy_situation_icon_frame_ui = 5 }
	}
	else_if = {
		limit = { 
			check_variable = { current_economy_situation_icon_frame < 0.70 }
		}
		set_variable = { current_economy_situation_icon_frame_ui = 6 }
	}
	else_if = {
		limit = { 
			check_variable = { current_economy_situation_icon_frame < 0.80 }
		}
		set_variable = { current_economy_situation_icon_frame_ui = 7 }
	}
	else_if = {
		limit = { 
			OR = {
				check_variable = { current_economy_situation_icon_frame < 1 }
				check_variable = { current_economy_situation_icon_frame = 1 }
			}
		}
		set_variable = { current_economy_situation_icon_frame_ui = 8 }
	}
}
economy_situation_decrease_2pc = {
	add_to_variable = { 
		var = current_economy_situation_icon_frame
		value = -0.02
		tooltip = tt_economy_situation_decrease_2pc
	}
}
